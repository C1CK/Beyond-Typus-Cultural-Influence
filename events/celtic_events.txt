country_event = { #Merges Gaelic and Brythonic groups
	id = cultural_influence_celtic.1
	title = goidelic_brythonic_merger.title
	desc = goidelic_brythonic_merger.desc
	picture = GREAT_BUILDING_eventPicture

	major = yes
	fire_only_once = yes

	trigger = {
		OR = {
			culture_group = gaelic
			culture_group = brythonic
		}
		total_development = 400 #2/3 England at game start
	}

	immediate = {
		hidden_effect = {
			merge_celtic = yes
			set_global_flag = gaels_and_britons_reunited_into_one_group
		}
	}

	option = {
		name = goidelic_brythonic_merger.a

		custom_tooltip = goidelic_brythonic_merger.tooltip
	}
}

province_event = { #Irish reclamation (Settle)
	id = cultural_influence_celtic.2
	title = irish_reclamation.title
	desc = irish_reclamation.desc
	picture = CULTURE_eventPicture

	trigger = {
		OR = {
			owner = { culture_group = gaelic }
			owner = { culture_group = celtic }
		}
		NOT = { culture = irish }
		NOT = { culture = celtic_irish }
		NOT = { culture = goidhealach }
		OR = { #Permit AI to settle accepted cultures since they don't understand
			has_owner_accepted_culture = no
			owner = { ai = yes }
		}
		OR = { #Intent check for beings capable of signaling intent (Requiring this allows lower MTTHs and hopefully more directed efforts)
			AND = { #AIs are dumb (With a lot of work it would be possible to use a system of decisions and flags to signal intent here)
				owner = { ai = yes }
				OR = {
					any_neighbor_province = { culture_group = celtic }
					any_neighbor_province = { culture_group = gaelic }
				}
			}
			num_of_times_improved_by_owner = 5 #Non-Dharma intent check
			AND = { #Dharma intent check
				has_colonist = yes
				is_city = yes
			}
		}
		#OR = { #Players can do it before the decision provided they dev it, or send a colonist
			#owner = { has_country_modifier = xxx } #No decision right now, but leaving here in-case it gets one
			#owner = { ai = no }
		#}
		gaelic_irish_culture_region = yes
		has_global_flag = CI_events_enabled
	}

	mean_time_to_happen = {
		#Template Settled MTTH
		years = 10

		#MTTH settings
		modifier = {
			factor = 0.5
			has_global_flag = 05_CI_MTTH_flag
		}
		modifier = {
			factor = 1.5
			has_global_flag = 15_CI_MTTH_flag
		}
		modifier = {
			factor = 2.0
			has_global_flag = 20_CI_MTTH_flag
		}
		modifier = {
			factor = 4.0
			has_global_flag = 40_CI_MTTH_flag
		}

		#Terrain factors (These replace pre-0.6.1 low dev penalties, which were simulating the same thing) (This was basically done to better simulate the German settlements in Banat)
		#Farmlands & Grasslands are ideal and act as the base 1.0 modifier (Marsh is also treated as Grasslands since it is difficult to tell if the Marsh has been drained)
		modifier = {
			factor = 2.5
			has_terrain = glacial
		}
		modifier = {
			factor = 2
			has_terrain = mountain
		}
		modifier = {
			factor = 1.5
			has_terrain = highlands
		}
		modifier = {
			factor = 1.5
			has_terrain = hills
		}
		modifier = {
			factor = 1.2
			has_terrain = woods
		}
		modifier = {
			factor = 1.2
			has_terrain = drylands
		}
		modifier = {
			factor = 1.4
			has_terrain = forest
		}
		modifier = {
			factor = 2
			has_terrain = desert
		}
		modifier = {
			factor = 1.8
			has_terrain = coastal_desert
		}
		#Climatic factors (Tropical ignored, as it isn't universally difficult to settle, only for the Temperate peoples)
		modifier = {
			has_climate = arid
			factor = 2
		}

		#Dev button factors
		modifier = {
			factor = 0.8
			num_of_times_improved_by_owner = 5
		}
		modifier = {
			factor = 0.8
			num_of_times_improved_by_owner = 10
		}

		#Dev Factors, penalties begin at 30 Dev
		modifier = {
			factor = 1.2
			development = 30
		}
		modifier = {
			factor = 1.2
			development = 40
		}
		modifier = {
			factor = 1.4
			development = 50
		}
		modifier = {
			factor = 1.4
			development = 60
		}
		modifier = {
			factor = 1.4
			development = 70
		}

		#Misc factors
		modifier = { #Passive settlement (I.E without a colonist)
			factor = 3
			has_colonist = no
		}
		modifier = {
			factor = 0.8
			has_idea_to_settle_faster = yes
		}
		modifier = {
			factor = 1.4
			has_owner_religion = no
		}
		modifier = {
			factor = 0.9
			adm = 4
		}
		modifier = {
			factor = 0.9
			adm = 5
		}
		modifier = {
			factor = 0.9
			adm = 6
		}
		#modifier = { #Don't like this, since misc got merged into BT adm focus will be active most of the time now for the advisor cost
		#	factor = 0.75
		#	national_focus = adm
		#}
	}

	option = {
		name = irish_reclamation.a

		if = {
			limit = {
				has_global_flag = gaels_and_britons_reunited_into_one_group
			}
			change_culture = celtic_irish
		}
		if = {
			limit = {
				NOT = { has_global_flag = gaels_and_britons_reunited_into_one_group }
			}
			change_culture = irish
		}
	}
}
