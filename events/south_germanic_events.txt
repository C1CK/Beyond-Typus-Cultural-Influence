province_event = { #Austrian Integration
	id = cultural_influence_germanic_rework.4
	title = austrian_integration.title
	desc = austrian_integration.desc
	picture = CULTURE_eventPicture
	goto = root

	trigger = {
		has_global_flag = hochdeutsch_created_flag
		culture = austrian
		NOT = { owner = { primary_culture = austrian } }
		OWNER = { #Is German
			OR = {
				primary_culture = hochdeutsch
				accepted_culture = hochdeutsch
			}
			culture_group = germanic
		}
		OR = { #Lack of Hochdeutsch influence or significant transit
			development = 20 
			province_trade_power = 10
			any_neighbor_province = {
				culture = hochdeutsch
			}
		}
		has_global_flag = CI_events_enabled
	}

	mean_time_to_happen = {
		years = 5 #No historic resistance, just not part of Germany
		modifier = { #Rural communities reluctant to convert
			factor = 1.2
			NOT = { development = 10 }
		}
		modifier = { #More extreme rural communities
			factor = 1.5
			NOT = { development = 5 }
		}
		modifier = { #Townships more connected to capital
			factor = 0.9
			development = 10
		}
		modifier = { #Cities even more connected
			factor = 0.8
			development = 20
		}
		modifier = { #Larger city means more connection
			factor = 0.7
			development = 30
		}
		modifier = { #Stop developing pls. k thx
			factor = 0.6
			development = 40
		}
	}

	option = {
		name = austrian_integration.a

		change_culture = hochdeutsch
	}
}

province_event = { #Dutch settlements and assimilation (Works for Low Franconian & Dutch)
	id = cultural_influence_germanic_rework.11
	title = low_franconian_settlements.title
	desc = {
		desc = low_franconian_settlements.desc
		trigger = { NOT = { owner = { primary_culture = dutch_modern } } }
	}
	desc = {
		desc = dutch_settlements.desc
		trigger = { owner = { primary_culture = dutch_modern } }
	}
	picture = CULTURE_eventPicture
	goto = root

	trigger = {
		owner = { culture_group = germanic }
		low_franconian_culture_region = yes #Use non-overlap region		
		NOT = { culture = dutch_modern }
		NOT = { culture = dutch } #Actually Low-Franconian
		NOT = { culture = plattduutsch }
		ci_settlement_check = yes
		OR = { #Players can do it before the decision provided they dev it, or send a colonist
			owner = { has_country_modifier = ostsiedlung }
			owner = { ai = no }
		}
		has_global_flag = CI_events_enabled
	}

	mean_time_to_happen = {
		#Template Settled MTTH
		years = 10

		#MTTH settings
		modifier = {
			factor = 0.0
			has_global_flag = 00_CI_MTTH_flag
		}		
		modifier = {
			factor = 0.5
			has_global_flag = 05_CI_MTTH_flag
		}
		modifier = {
			factor = 1.5
			has_global_flag = 15_CI_MTTH_flag
		}
		modifier = {
			factor = 2.0
			has_global_flag = 20_CI_MTTH_flag
		}
		modifier = {
			factor = 4.0
			has_global_flag = 40_CI_MTTH_flag
		}

		#Terrain factors (These replace pre-0.6.1 low dev penalties, which were simulating the same thing) (This was basically done to better simulate the German settlements in Banat)
		#Farmlands & Grasslands are ideal and act as the base 1.0 modifier (Marsh is also treated as Grasslands since it is difficult to tell if the Marsh has been drained)
		modifier = {
			factor = 2.5
			has_terrain = glacier
