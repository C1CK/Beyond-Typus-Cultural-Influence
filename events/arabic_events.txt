province_event = { #Arabic adoption (Masri)
	id = cultural_influence_arabic.3
	title = masri_adoption.title
	desc = masri_adoption.desc
	picture = CULTURE_eventPicture
	goto = root
	
	trigger = {
		OR = {
			owner = { culture_group = turko_semitic }
			AND = { #If you accept the Arabic Dialect and it isn't in your group, you can Arabicize the province
				owner = { accepted_culture = al_misr_arabic }
				NOT = { culture_group = OWNER }
			}
		}
		masri_culture_region = yes			
		NOT = { culture = al_misr_arabic }
		NOT = { culture = mashriqi }
		ci_assimilation_check = yes
		has_global_flag = CI_events_enabled
	}

	mean_time_to_happen = { #Generic Assimilation MTTH
		years = 25

		#MTTH settings
		modifier = {
			factor = 0.0
			has_global_flag = 00_CI_MTTH_flag
		}
		modifier = {
			factor = 0.5
			has_global_flag = 05_CI_MTTH_flag
		}
		modifier = {
			factor = 1.5
			has_global_flag = 15_CI_MTTH_flag
		}
		modifier = {
			factor = 2.0
			has_global_flag = 20_CI_MTTH_flag
		}
		modifier = {
			factor = 4.0
			has_global_flag = 40_CI_MTTH_flag
		}

		modifier = {
			factor = 0.5
			culture_group = turko_semitic #Arabic
		}

		#Dev. factors
		modifier = {
			factor = 1.2
			NOT = { development = 20 }
		}
		modifier = {
			factor = 1.2
			development = 30
			NOT = { has_building = university }
		}
		modifier = {
			factor = 1.4
			development = 40
			NOT = { has_building = university }
		}
		modifier = {
			factor = 1.4
			development = 50
			NOT = { has_building = university }
		}
		modifier = { #This is now some 80 years, don't see much reason to increment further
			factor = 1.4
			development = 60
			NOT = { has_building = university }
		}

		#Religion
		modifier = {
			factor = 1.4
			NOT = { tolerance_to_this = 0 }
		}
		modifier = {
			factor = 1.1
			has_owner_religion = no
		}
		modifier = {
			factor = 1.1
			religion_group = OWNER
		}

		#Misc
		modifier = {
			factor = 1.2
			unrest = 1
		}
		modifier = {
			factor = 0.9
			has_owner_accepted_culture = yes
		}
		modifier = {
			factor = 0.8
			has_building = university
		}
		modifier = {
			factor = 0.9
			has_colonist = yes
		}
		modifier = {
			factor = 1.25
			is_in_capital_area = no
		}
	}

	option = {
		name = masri_adoption.a

		if = {
			limit = {
				has_global_flag = mashriqi_created_flag
			}
			change_culture = mashriqi
		}
		if = {
			limit = {
				NOT = { has_global_flag = mashriqi_created_flag }
			}
			change_culture = al_misr_arabic
		}
	}
}
