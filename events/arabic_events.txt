province_event = { #Arabic adoption (Masri)
	id = cultural_influence_arabic.3
	title = masri_adoption.title
	desc = masri_adoption.desc
	picture = CULTURE_eventPicture

	trigger = {
		owner = { culture_group = turko_semitic }
		NOT = { culture = al_misr_arabic }
		NOT = { culture = mashriqi }
		OR = { #Permit AI to settle accepted cultures since they don't understand
			has_owner_accepted_culture = no
			owner = { ai = yes }
		}
		religion_group = muslim
		OR = {
			OR = {
			development = 10 #To prevent Sahara converting
			culture_group = turko_semitic
		}
			culture_group = turko_semitic
		}
		OR = {
			any_neighbor_province = { culture_group = turko_semitic }
			province_trade_power = 15 #10 from level 2 CoT vanilla, so 15 seems fair, to require a building as well
			is_capital = yes
		}
		masri_culture_region = yes
		has_global_flag = CI_events_enabled
	}

	mean_time_to_happen = {
		years = 25

		modifier = {
			factor = 0.5
			has_global_flag = 05_CI_MTTH_flag
		}
		modifier = {
			factor = 1.5
			has_global_flag = 15_CI_MTTH_flag
		}
		modifier = {
			factor = 2.0
			has_global_flag = 20_CI_MTTH_flag
		}
		modifier = {
			factor = 4.0
			has_global_flag = 40_CI_MTTH_flag
		}

		modifier = { #Since a lot of the land is uncolonized
			factor = 0.1
			culture_group = turko_semitic
		}

		modifier = {
			factor = 0.8
			has_idea_to_settle_faster = yes
		}
		modifier = {
			factor = 0.8
			NOT = { development = 20 }
		}
		modifier = {
			factor = 1.2
			development = 30
		}
		modifier = {
			factor = 1.4
			development = 40
		}
