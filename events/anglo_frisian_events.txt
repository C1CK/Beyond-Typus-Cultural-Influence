namespace = cultural_influence_anglo_frisian

province_event = { #Frisian Splinter
	id = cultural_influence_anglo_frisian.1
	title = frisian_split.title
	desc = frisian_split.desc
	picture = CULTURE_eventPicture
	
	trigger = {
		is_year = 1500
		culture = frisian
		NOT = { owner = { culture_group = anglo_frisian } }
		NOT = { any_country = { primary_culture = frisian } }
		NOT = { has_global_flag = frisian_split }
		nationalism = 0
	}
	
	mean_time_to_happen = {
		years = 1
	}
	
	immediate = {
		hidden_effect = {
			set_global_flag = frisian_split
		}	
	}

	option = {
		name = frisian_split.a

		every_province = {
			limit = {
				culture = frisian
				north_frisian_culture_region = yes
			}
			change_culture = north_frisian
		}
		every_province = {
			limit = {
				culture = frisian
				west_frisian_culture_region = yes
			}
			change_culture = west_frisian
		}
	}
}

province_event = { #West Frisian settlements (Settle)
	id = cultural_influence_anglo_frisian.2
	title = west_frisian_settlements.title
	desc = west_frisian_settlements.desc
	picture = CULTURE_eventPicture
	goto = root
	
	trigger = {
		west_frisian_culture_region = yes	
		owner = { culture_group = anglo_frisian }	
		NOT = { culture = west_frisian }
		NOT = { culture = frisian }
		ci_settlement_check = yes	#Intent check for beings capable of signaling intent (Requiring this allows lower MTTHs and hopefully more directed efforts)
		OR = { #Players can do it before the decision provided they dev it, or send a colonist
			owner = { has_country_modifier = frisian_settlements }
			owner = { ai = no }
		}
		has_global_flag = CI_events_enabled
	}

	mean_time_to_happen = {
		#Template Settled MTTH
		years = 10

		#MTTH settings
		modifier = {
			factor = 0.0
			has_global_flag = 00_CI_MTTH_flag
		}		
		modifier = {
			factor = 0.5
			has_global_flag = 05_CI_MTTH_flag
		}
		modifier = {
			factor = 1.5
			has_global_flag = 15_CI_MTTH_flag
		}
		modifier = {
			factor = 2.0
			has_global_flag = 20_CI_MTTH_flag
		}
		modifier = {
			factor = 4.0
			has_global_flag = 40_CI_MTTH_flag
		}

		#Terrain factors (These replace pre-0.6.1 low dev penalties, which were simulating the same thing) (This was basically done to better simulate the German settlements in Banat)
		#Farmlands & Grasslands are ideal and act as the base 1.0 modifier (Marsh is also treated as Grasslands since it is difficult to tell if the Marsh has been drained)
		modifier = {
			factor = 2.5
			has_terrain = glacier
		}
		modifier = {
			factor = 2
			OR = {
				has_terrain = mountain
				has_terrain = desert
			}	
		}
		modifier = {
			factor = 1.5
			OR = {
				has_terrain = highlands
				has_terrain = hills
			}	
		}
		modifier = {
			factor = 1.2
			OR = { 
				has_terrain = woods
				has_terrain = drylands
			}	
		}
		modifier = {
			factor = 1.4
			has_terrain = forest
		}
		modifier = {
			factor = 1.8
			has_terrain = coastal_desert
		}
		#Climatic factors (Tropical ignored, as it isn't universally difficult to settle, only for the Temperate peoples)
		modifier = {
			has_climate = arid
			factor = 2
		}

		#Dev button factors
		modifier = {
			factor = 0.8
			num_of_times_improved_by_owner = 5
		}
		modifier = {
			factor = 0.8
			num_of_times_improved_by_owner = 10
		}

		#Dev Factors, penalties begin at 30 Dev
		modifier = {
			factor = 1.2
			development = 30
		}
		modifier = {
			factor = 1.2
			development = 40
		}
		modifier = {
			factor = 1.4
			development = 50
		}
		modifier = {
			factor = 1.4
			development = 60
		}
		modifier = {
			factor = 1.4
			development = 70
		}

		#Misc factors
		modifier = { #Passive settlement (I.E without a colonist)
			factor = 3
			has_colonist = no
		}
		modifier = {
			factor = 0.8
			owner = { has_idea_to_settle_faster = yes }
		}
		modifier = {
			factor = 1.4
			has_owner_religion = no
		}
		modifier = {
			factor = 0.9
			owner = { adm = 4 }
		}
		modifier = {
			factor = 0.9
			owner = { adm = 5 }
		}
		modifier = {
			factor = 0.9
			owner = { adm = 6 }
		}
		#modifier = { #Don't like this, since misc got merged into BT adm focus will be active most of the time now for the advisor cost
		#	factor = 0.75
		#	national_focus = adm
		#}
	}

	option = {
		name = west_frisian_settlements.a

		if = {
			limit = {
				has_global_flag = frisian_split
			}
			change_culture = west_frisian
		}
		if = {
			limit = {
				NOT = { has_global_flag = frisian_split }
			}
			change_culture = frisian
		}
	}
}
