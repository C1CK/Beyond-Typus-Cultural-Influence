province_event = { #French expansion (Home Region)
	id = cultural_influence_french.1
	title = french_expansion.title
	desc = french_expansion.desc
	picture = RELIGIOUS_CONVERSION_eventPicture
	goto = root
	
	trigger = {
		owner = { culture_group = french }
		NOT = { culture = OWNER }
		OR = { #Permit AI to settle accepted cultures since they don't understand
			AND = {
				has_owner_accepted_culture = no
				NOT = { culture_group = french }
			}	
			owner = { ai = yes }
		}
		OR = {
			any_neighbor_province = { culture = OWNER }
			any_friendly_coast_border_province = { culture = OWNER } #This is to fix the channel islands not flipping basically
		}
		owner = {
			num_of_owned_provinces_with = {
				value = 45 #At least ~2/3 of France's 71 provinces
				is_state = yes
				french_homeland = yes
			}
		}
		french_homeland = yes
		printing_press = 100 #Check if province has institution not country
		has_global_flag = CI_events_enabled
	}

	mean_time_to_happen = {
		years = 20
		#MTTH Modifiers
		modifier = {
			factor = 0.0
			has_global_flag = 00_CI_MTTH_flag
		}
		modifier = {
			factor = 0.5
			has_global_flag = 05_CI_MTTH_flag
		}
		modifier = {
			factor = 1.5
			has_global_flag = 15_CI_MTTH_flag
		}
		modifier = {
			factor = 2.0
			has_global_flag = 20_CI_MTTH_flag
		}
		modifier = {
			factor = 4.0
			has_global_flag = 40_CI_MTTH_flag
		}
		modifier = {
			factor = 0.9
			development = 20
		}
		modifier = {
			factor = 0.9
			development = 30
		}
		modifier = {
			factor = 0.9
