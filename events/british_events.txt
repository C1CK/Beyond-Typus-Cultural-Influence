country_event = { #Northumbrian -> English / English -> Northumbrian
	id = cultural_influence_british.1
	title = british_first_unification.title
	desc = british_first_unification.desc
	picture = ARTWORK_eventPicture

	major = yes
	fire_only_once = yes

	trigger = {
		government_rank = 2
		culture_group = british
		NOT = { has_global_flag = british_first_unification_flag }
		has_global_flag = scottish_solidified_flag
		europe = { #Owned by ROOT, or not needed
			OR = {
				AND = {
					NOT = { culture = english }
					NOT = { culture = northumbrian }
				}
				owned_by = ROOT
			}
			type = all
		}
		NOT = { #Invert
			any_province = { #Any province that is English/Inglish lacks printing press
				continent = europe
				OR = {
					culture = english
					culture = northumbrian #Inglish
				}
				NOT = { printing_press = 100 }
			}
		}
		OR = {
			religion = anglican #Prayer Book
			has_institution = printing_press
		}
	}

	mean_time_to_happen = {
		months = 6
	}

	immediate = {
		hidden_effect = {
			set_global_flag = british_first_unification_flag
		}
	}

	option = { #Southern
		name = british_first_unification.a

		trigger = { primary_culture = english }

		custom_tooltip = british_first_unification_english.effect
		hidden_effect = {
			every_province = {
				limit = {
					OR = {
						culture = english
						culture = northumbrian
					}
				}
				change_culture = english
			}
		}
		239 = { #Gloucestershire
			add_base_tax = 2
		}
		capital_scope = { #Capital
			add_base_tax = 2
		}
		if = { #House of York
			limit = {
				dynasty = York
			}
			4367 = { #York
				add_base_tax = 4
			}
		}
		if = { #House of Lancaster
			limit = {
				dynasty = Lancaster
			}
			244 = { #Lancaster
				add_base_tax = 4
			}
		}
		if = { #If this is you, you're probably a Tudor 
			limit = {

		246 = { #Northumberland
			add_base_tax = 2
		}
		243 = { #Lincolnshire
			add_base_tax = 2
		}

			add_base_tax = 2

country_event = { #Solidifying of Scots
	id = cultural_influence_british.2
	title = solidifying_of_scots.title
	desc = solidifying_of_scots.desc
	picture = ARTWORK_eventPicture

	fire_only_once = yes
	major = yes

	trigger = {
		is_year = 1480 #Simply gating this by year for now
		culture_group = british
		OR = { #Ensure it fires for the correct winner (If stalemate goes to one of the two at random)
			owns_core_province = 248 #Lothian
			owns_core_province = 246 #Northumberland
		}
		NOT = { SCO = { war_with = ENG } }
		NOT = { NOL = { war_with = SCO } }
		NOT = { NOL = { war_with = ENG } }
	}

	mean_time_to_happen = {
		years = 5
	}

	option = { #Stalemate
		name = solidifying_of_scots.a

		trigger = {
			248 = { NOT = { OWNER = { primary_culture = english } } } #Lothian
			248 = { NOT = { OWNER = { primary_culture = northumbrian } } } #Lothian
			246 = { NOT = { OWNER = { primary_culture = scottish } } } #Northumberland
		}

		set_global_flag = scottish_solidified_flag
		#melt_into_modern_english = yes #Will be added come 1291 start

		SCO = { remove_historical_rival = ENG }
		ENG = { remove_historical_rival = SCO }
		#Shift based on OWNER if neither is dominant
		british_isles_region = { #Shift Scots outside of Scotland -> Northumbrian
			limit = {
				anglo_northumbrian_culture_region = yes
				culture = scottish
				OR = {
					OWNER = { primary_culture = english }
					OWNER = { primary_culture = northumbrian }
				}
			}
			change_culture = northumbrian
		}
	}

	option = { #England dominant
		name = solidifying_of_scots.b

		trigger = {
			OR = {
				248 = { OWNER = { primary_culture = english } } #Lothian
				248 = { OWNER = { primary_culture = northumbrian } } #Lothian
			}
		}

		set_global_flag = scottish_solidified_flag
		#melt_into_modern_english = yes #Will be added come 1291 start

		add_prestige = 25
		#Shift it all if England is dominant
		british_isles_region = { #Shift Scots outside of Scotland -> Northumbrian
			limit = {
				anglo_northumbrian_culture_region = yes
				culture = scottish
				#NOT = { OWNER = { primary_culture = scottish } }
			}
			change_culture = northumbrian
		}
	}

	option = { #Scotland dominant
		name = solidifying_of_scots.c

		trigger = {
			246 = { OWNER = { primary_culture = scottish } } #Northumberland
		}

		set_global_flag = scottish_solidified_flag
		#melt_into_modern_english = yes #Will be added come 1291 start

		add_prestige = 25
		#Don't shift anything if Scotland is dominant
		#british_isles_region = { #Shift Scots outside of Scotland -> Northumbrian
		#	limit = {
		#		anglo_northumbrian_culture_region = yes
		#		culture = scottish
		#		NOT = { OWNER = { primary_culture = scottish } }
		#	}
		#	change_culture = northumbrian
		#}
	}
}

	}

	mean_time_to_happen = {
		years = 50	#50 years for native cultures to get settled
		#MTTH settings
		modifier = {
			factor = 0.0
			has_global_flag = 00_CI_MTTH_flag
		}
		modifier = {
			factor = 0.5
			has_global_flag = 05_CI_MTTH_flag
		}
		modifier = {
			factor = 1.5
			has_global_flag = 15_CI_MTTH_flag
		}
		modifier = {
			factor = 2.0
			has_global_flag = 20_CI_MTTH_flag
		}
		modifier = {
			factor = 4.0
			has_global_flag = 40_CI_MTTH_flag
		}
		modifier = {	#About 15 years for colonies to diverge (this is after the initial establishment which results in about 85 years total for the earlier provinces)
			factor = 0.30
			british_european = yes
		}
		modifier = {
			factor = 0.15	#About 7.5 years if a separate colonial culture
			british_new_world = yes
		}
		modifier = {	#Once the first provinces flip neighbours will flip faster
			factor = 0.5
			any_neighbor_province = { culture = anglo_australian }
		}
		modifier = {	#If independence is achieved divergence is quicker
			factor = 0.4
			AND = {
				culture_group = british
				owner = { is_former_colonial_nation = yes }
			}	
		}
	}
	immediate = {
		hidden_effect = {
			every_country = {
				limit = {
					culture_group = british
					capital_scope = { #This is a fallback trigger in case any new tags spawn after the initial establishment event
						anglo_australian_culture_region = yes
					}
				}
				change_primary_culture = anglo_australian 
			}
		}
	}

	option = {
		name = british_australian.a

		change_culture = anglo_australian
		if = {
			limit = { OWNER = { NOT = { accepted_culture = anglo_australian } } }
			add_accepted_culture = anglo_australian
		}			#Adding culture as an accepted culture to prevent AI flipping, and to not penalise player in a no-choice culture event		
	}
}

			AND = {	#Or if for some reason the US doesn't form but there is an independent colonial tag
				capital_scope = {
					anglo_american_culture_region = yes
				}
				is_former_colonial_nation = yes
			}
		}
		NOT = {	#Basically checking if the Americans are under foreign control, if they are, they don't split.
			OR = {
				AND = { #Provinces fell to Foreign rule
					calc_true_if = {
						all_province = {
							culture = american
							owner = { NOT = { primary_culture = american } }
						}
						amount = 25
					}
				}
				953 = { #Checking if Washington is under foreign domination (Principally checking for a British Colony)
					OWNER = {
						culture_group = british
						is_subject = yes
						OVERLORD = {
							NOT = { primary_culture = american }
						}
					}
				}
			}
		}
		has_global_flag = CI_events_enabled
		NOT = { has_global_flag = american_splinter_flag }
	}

	mean_time_to_happen = {
		years = 75

		#MTTH settings
		modifier = {
			factor = 0.0
			has_global_flag = 00_CI_MTTH_flag
		}

			factor = 0.5

			factor = 4.0

			set_global_flag = american_splinter_flag

			every_country = {
				limit = {
					primary_culture = american
				}

					id = cultural_influence_british.62 #Pick new Primary

	}

province_event = { #Settlement of the Frontier
	id = cultural_influence_british.80
	title = british_frontier_settlement.title
	desc = british_frontier_settlement.desc
	picture = CULTURE_eventPicture
	
	trigger = {
		OR = {
			native_american_trigger = yes #Native cultures
			is_capital = yes
		}	
		owner = { culture_group = british }
		british_european_country = yes	#Once the colonial language is established they will use their own events instead, if it's a crown colony then either event can fire, eventually it will flip to the colonial varient
		ci_colonial_settlement_check = yes
	}

	mean_time_to_happen = {
		#Template Settled MTTH
		years = 50

		modifier = {
			factor = 0.8
			owner = { has_idea_to_settle_faster = yes }
		}
		modifier = {
			factor = 0.8
			OR = {
				owner = { is_colonial_nation = yes }
				owner = { is_former_colonial_nation = yes }
			}
		}
		modifier = {
			factor = 0.6
			any_neighbor_province = {
				culture = OWNER
			}
		}
		modifier = {
			factor = 0.6
			religion = OWNER
		}
		modifier = {
			factor = 0.6
			is_capital = yes
		}
	}

	option = {
		name = british_frontier_settlement.a

		change_culture = OWNER
		if = {
			limit = {
				NOT = { religion_group = OWNER }
			}
			change_religion = OWNER
		}	
	}
}
