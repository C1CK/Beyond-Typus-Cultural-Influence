country_event = { #Christian Granada Latinizes
	id = cultural_influence_iberian.1
	title = christian_moors.title
	desc = christian_moors.desc
	picture = RELIGIOUS_CONVERSION_eventPicture

	major = yes

	trigger = {
		culture = andalusi_culture
		capital_scope = { region = iberia_region }
		religion_group = christian
		NOT = { has_country_flag = ci_declined_latinization }
	}

	mean_time_to_happen = {
		months = 30
	}

	option = { #Latinize
		name = christian_moors.a

		#Subject ifs
		if = { #Iberian subject
			limit = {
				is_subject = yes
				overlord = {
					culture_group = iberian
				}
			}
			change_primary_culture = andalucian
			capital_scope = { change_culture = andalucian }
		}
		if = { #Occitan subject
			limit = {
				is_subject = yes
				overlord = {
					culture_group = occitanic
				}
			}
			change_primary_culture = occitan_andalucian
			capital_scope = { change_culture = occitan_andalucian }
		}

		#Indep. ifs
		if = { #Iberian dominant
			limit = {
				is_subject = no
				NOT = { has_global_flag = ci_restoration_of_aragonese_flag }
			}
			change_primary_culture = andalucian
			capital_scope = { change_culture = andalucian }
		}
		if = { #Occitan dominant
			limit = {
				is_subject = no
				has_global_flag = ci_restoration_of_aragonese_flag
			}
			change_primary_culture = occitan_andalucian
			capital_scope = { change_culture = occitan_andalucian }
		}

		ai_chance = {
			factor = 1
		}
	}

	option = { #No
		name = christian_moors.b

		add_prestige = 10
		set_country_flag = ci_declined_latinization
	}
}

province_event = { #Castillian expansion (Home Region)
	id = cultural_influence_iberian.2
	title = iberian_expansion.title
	desc = iberian_expansion.desc
	picture = RELIGIOUS_CONVERSION_eventPicture
	goto = root
	
	trigger = {
		owner = { primary_culture = castillian }
		NOT = { nationalism = 1 }
		NOT = { culture = catalan } #Catalan exempted until I can think of a better way to represent their historic resiliance
		NOT = { culture = castillian }
		OR = { #Permit AI to settle accepted cultures since they don't understand
			AND = {
				has_owner_accepted_culture = no
				NOT = { culture_group = iberian }
			}	
			owner = { ai = yes }
		}
		OR = {
			any_neighbor_province = {
				culture = castillian
			}
			any_friendly_coast_border_province = {
				culture = castillian
			}
			is_island = yes
			is_capital = yes
		}
		iberian_homeland = yes
		has_global_flag = CI_events_enabled
	}

	mean_time_to_happen = { #Generic Italic Assimilation MTTH
		years = 20

		#MTTH settings
		modifier = {
			factor = 0.0
			has_global_flag = 00_CI_MTTH_flag
		}
		modifier = {
			factor = 0.5
			has_global_flag = 05_CI_MTTH_flag
		}
		modifier = {
			factor = 1.5
			has_global_flag = 15_CI_MTTH_flag
		}
		modifier = {
			factor = 2.0
			has_global_flag = 20_CI_MTTH_flag
		}
		modifier = {
			factor = 4.0
			has_global_flag = 40_CI_MTTH_flag
		}

		#Remote land
		modifier = {
			factor = 1.5
			NOT = { development = 10 }
		}
		modifier = {
			factor = 1.25
			NOT = { development = 20 }
		}

		#Terrain penalties (Basque and Aragonese retreat to the Mountain villages)
		modifier = {
			factor = 1.5
			has_terrain = highlands
		}
		modifier = {
			factor = 2
			has_terrain = hills
		}
		modifier = {
			factor = 2.5
			has_terrain = mountain
		}

		#Jumping
		modifier = {
			factor = 1.5
			NOT = {
				any_neighbor_province = {
					culture = castillian
				}
			}
		}
		modifier = {
			factor = 1.5
			is_island = yes
		}

		#Weak Influence
		modifier = {
			factor = 1.2
			NOT = {
				calc_true_if = {
					all_neighbor_province = {
						culture = castillian
					}
					amount = 2
				}
			}
		}

		#Strong Influence
		modifier = {
			factor = 0.8
			calc_true_if = {
				all_neighbor_province = {
					culture = castillian
				}
				amount = 2
			}
		}
		modifier = {
			factor = 0.8
			calc_true_if = {
				all_neighbor_province = {
					culture = castillian
				}
				amount = 3
			}
		}
	}

	option = {
		name = iberian_expansion.a

		change_culture = castillian
	}
}

namespace = cultural_influence_iberian_colonial

country_event = { #Establishment of Brasileiro
	id = cultural_influence_iberian_colonial.1
	title = luso_brazilian_establishment.title
	desc = luso_brazilian_establishment.desc
	picture = CULTURE_eventPicture
	
	trigger = {
		any_owned_province = {
			AND = {
				ibero_brazilian_culture_region = yes
				lusophone_cultures = yes
			}	
		}
		culture_group = iberian
		NOT = { has_global_flag = luso_brazilian_established_flag }
	}
	mean_time_to_happen = {
		years = 70
	}
	
	immediate = {
		hidden_effect = {
			set_global_flag = luso_brazilian_established_flag
			every_country = {
				limit = {
					lusophone_cultures_country = yes
					capital_scope = { ibero_brazilian_culture_region = yes }
				}
				capital_scope = { change_culture = brazilian }
				change_primary_culture = brazilian
			}
		}
	}
	
	option = {
		name = luso_brazilian_establishment.a
		custom_tooltip = luso_brazilian_established_convert_tags
	}
}

province_event = { #Brasileiro Colonial settlement event
	id = cultural_influence_iberian_colonial.2
	title = luso_brazilian_settlement.title
	desc = luso_brazilian_settlement.desc
	picture = CULTURE_eventPicture
	goto = root
	
	trigger = {
		ibero_brazilian_culture_region = yes
		NOT = { culture = brazilian }
		is_city = yes
		OR = {	#If Lusophone, allow converting other non Lusophone Iberian provinces, otherwise only convert Lusophone provinces
			OWNER = { lusophone_cultures_country = yes }
			AND = {
				OWNER = { culture_group = iberian }
				OWNER = { lusophone_cultures_country = no }
				lusophone_cultures = yes
			}	
		}	
		any_neighbor_province = { lusophone_cultures = yes }		
		ci_colonial_settlement_check = yes
		has_global_flag = luso_brazilian_established_flag
	}

	mean_time_to_happen = {
		years = 50	#50 years for native cultures to get settled
		#MTTH settings
		modifier = {
			factor = 0.0
			has_global_flag = 00_CI_MTTH_flag
		}
		modifier = {
			factor = 0.5
			has_global_flag = 05_CI_MTTH_flag
		}
		modifier = {
			factor = 1.5
			has_global_flag = 15_CI_MTTH_flag
		}
		modifier = {
			factor = 2.0
			has_global_flag = 20_CI_MTTH_flag
		}
		modifier = {
			factor = 4.0
			has_global_flag = 40_CI_MTTH_flag
		}
		modifier = {	#About 15 years for colonies to diverge (this is after the initial establishment which results in about 85 years total for the earlier provinces)
			factor = 0.30
			iberian_old_worlder = yes
		}
		modifier = {
			factor = 0.15	#About 7.5 years if a separate colonial culture
			iberian_new_worlder = yes
		}
		modifier = {	#Once the first provinces flip neighbours will flip faster
			factor = 0.5
			any_neighbor_province = { culture = brazilian }
		}
		modifier = {	#If independence is achieved divergence is quicker
			factor = 0.4
			AND = {
				culture_group = iberian
				owner = { is_former_colonial_nation = yes }
			}	
		}
	}

	immediate = {
		hidden_effect = {
			every_country = {
				limit = {
					lusophone_cultures_country = yes
					capital_scope = { #This is a fallback trigger in case any new tags spawn after the initial establishment event
						ibero_brazilian_culture_region = yes
					}
				}
				change_primary_culture = brazilian 
			}
		}
	}

	option = {
		name = luso_brazilian_settlement.a

		change_culture = brazilian
		if = {
			limit = { OWNER = { NOT = { accepted_culture = brazilian } } }
			add_accepted_culture = brazilian
		}			#Adding culture as an accepted culture to prevent AI flipping, and to not penalise player in a no-choice culture event		
	}
}

province_event = { #Luso-Centroamericano Colonial settlement event
	id = cultural_influence_iberian_colonial.4
	title = luso_central_american_settlement.title
	desc = luso_central_american_settlement.desc
	picture = CULTURE_eventPicture
	goto = root
	
	trigger = {
		ibero_central_american_culture_region = yes
		NOT = { culture = portuguese_central_american }
		is_city = yes
		OR = {	#If Lusophone, allow converting other non Lusophone Iberian provinces, otherwise only convert Lusophone provinces
			OWNER = { lusophone_cultures_country = yes }
			AND = {
				OWNER = { culture_group = iberian }
				OWNER = { lusophone_cultures_country = no }
				lusophone_cultures = yes
			}	
		}	
		any_neighbor_province = { lusophone_cultures = yes }		
		ci_colonial_settlement_check = yes
		has_global_flag = luso_central_american_established_flag
	}

	mean_time_to_happen = {
		years = 50	#50 years for native cultures to get settled
		#MTTH settings
		modifier = {
			factor = 0.0
			has_global_flag = 00_CI_MTTH_flag
		}
		modifier = {
			factor = 0.5
			has_global_flag = 05_CI_MTTH_flag
		}
		modifier = {
			factor = 1.5
			has_global_flag = 15_CI_MTTH_flag
		}
		modifier = {
			factor = 2.0
			has_global_flag = 20_CI_MTTH_flag
		}
		modifier = {
			factor = 4.0
			has_global_flag = 40_CI_MTTH_flag
		}
		modifier = {	#About 15 years for colonies to diverge (this is after the initial establishment which results in about 85 years total for the earlier provinces)
			factor = 0.30
			iberian_old_worlder = yes
		}
		modifier = {
			factor = 0.15	#About 7.5 years if a separate colonial culture
			iberian_new_worlder = yes
		}
		modifier = {	#Once the first provinces flip neighbours will flip faster
			factor = 0.5
			any_neighbor_province = { culture = portuguese_central_american }
		}
		modifier = {	#If independence is achieved divergence is quicker
			factor = 0.4
			AND = {
				culture_group = iberian
				owner = { is_former_colonial_nation = yes }
			}	
		}
	}

	immediate = {
		hidden_effect = {
			every_country = {
				limit = {
					lusophone_cultures_country = yes
					capital_scope = { #This is a fallback trigger in case any new tags spawn after the initial establishment event
						ibero_central_american_culture_region = yes
					}
				}
				change_primary_culture = portuguese_central_american 
			}
		}
	}

	option = {
		name = luso_central_american_settlement.a

		change_culture = portuguese_central_american
		if = {
			limit = { OWNER = { NOT = { accepted_culture = portuguese_central_american } } }
			add_accepted_culture = portuguese_central_american
		}			#Adding culture as an accepted culture to prevent AI flipping, and to not penalise player in a no-choice culture event		
	}
}

	trigger = {
		ibero_canadian_culture_region = yes
		NOT = { culture = portuguese_canadian }
		is_city = yes
		OR = {	#If Lusophone, allow converting other non Lusophone Iberian provinces, otherwise only convert Lusophone provinces
			OWNER = { lusophone_cultures_country = yes }
			AND = {
				OWNER = { culture_group = iberian }
				OWNER = { lusophone_cultures_country = no }
				lusophone_cultures = yes
			}	
		}	
		any_neighbor_province = { lusophone_cultures = yes }		
		ci_colonial_settlement_check = yes
		has_global_flag = luso_canadian_established_flag
	}

	mean_time_to_happen = {
		years = 50	#50 years for native cultures to get settled
		#MTTH settings
		modifier = {
			factor = 0.0
			has_global_flag = 00_CI_MTTH_flag
		}
		modifier = {
			factor = 0.5
			has_global_flag = 05_CI_MTTH_flag
		}
		modifier = {
			factor = 1.5
			has_global_flag = 15_CI_MTTH_flag
		}
		modifier = {
			factor = 2.0
			has_global_flag = 20_CI_MTTH_flag
		}
		modifier = {
			factor = 4.0
			has_global_flag = 40_CI_MTTH_flag
		}
		modifier = {	#About 15 years for colonies to diverge (this is after the initial establishment which results in about 85 years total for the earlier provinces)
			factor = 0.30
			iberian_old_worlder = yes
		}
		modifier = {
			factor = 0.15	#About 7.5 years if a separate colonial culture
			iberian_new_worlder = yes
		}
		modifier = {	#Once the first provinces flip neighbours will flip faster
			factor = 0.5
			any_neighbor_province = { culture = portuguese_canadian }
		}
		modifier = {	#If independence is achieved divergence is quicker
			factor = 0.4
			AND = {
				culture_group = iberian
				owner = { is_former_colonial_nation = yes }
			}	
		}
	}

	immediate = {
		hidden_effect = {
			every_country = {
				limit = {
					lusophone_cultures_country = yes
					capital_scope = { #This is a fallback trigger in case any new tags spawn after the initial establishment event
						ibero_canadian_culture_region = yes
					}
				}
				change_primary_culture = portuguese_canadian 
			}
		}
	}

	option = {
		name = luso_canadian_settlement.a

		change_culture = portuguese_canadian
		if = {
			limit = { OWNER = { NOT = { accepted_culture = portuguese_canadian } } }
			add_accepted_culture = portuguese_canadian
		}			#Adding culture as an accepted culture to prevent AI flipping, and to not penalise player in a no-choice culture event		
	}
}

		modifier = {
			factor = 0.15	#About 7.5 years if a separate colonial culture
			iberian_new_worlder = yes
		}
		modifier = {	#Once the first provinces flip neighbours will flip faster
			factor = 0.5
			any_neighbor_province = { culture = spanish_central_american }
		}

			factor = 0.4

				change_primary_culture = spanish_central_american

	}

		name = spanish_central_american_settlement.a

		change_culture = spanish_central_american
		if = {
			limit = { OWNER = { NOT = { accepted_culture = spanish_central_american } } }

		}			#Adding culture as an accepted culture to prevent AI flipping, and to not penalise player in a no-choice culture event		
	}

	option = {
		name = spanish_south_african_settlement.a

		change_culture = spanish_south_african
		if = {
			limit = { OWNER = { NOT = { accepted_culture = spanish_south_african } } }
			add_accepted_culture = spanish_south_african
		}			#Adding culture as an accepted culture to prevent AI flipping, and to not penalise player in a no-choice culture event		
	}

province_event = { #Sul-Africano Colonial settlement event
	id = cultural_influence_iberian_colonial.72
	title = luso_south_african_settlement.title
	desc = luso_south_african_settlement.desc
	picture = CULTURE_eventPicture
	goto = root
	
	trigger = {
		ibero_south_african_culture_region = yes
		NOT = { culture = portuguese_south_african }
		is_city = yes
		OR = {	#If Hispanophone, allow converting other non Hispanophone Iberian provinces, otherwise only convert Hispanophone provinces
			OWNER = { lusophone_cultures_country = yes }
			AND = {
				OWNER = { culture_group = iberian }
				OWNER = { hispanophone_cultures_country = no }
				lusophone_cultures = yes
			}	
		}	
		any_neighbor_province = { lusophone_cultures = yes }		
		ci_colonial_settlement_check = yes
		has_global_flag = luso_south_african_established_flag
	}

	mean_time_to_happen = {
		years = 50	#50 years for native cultures to get settled
		#MTTH settings
		modifier = {
			factor = 0.0
			has_global_flag = 00_CI_MTTH_flag
		}
		modifier = {
			factor = 0.5
			has_global_flag = 05_CI_MTTH_flag
		}
		modifier = {
			factor = 1.5
			has_global_flag = 15_CI_MTTH_flag
		}
		modifier = {
			factor = 2.0
			has_global_flag = 20_CI_MTTH_flag
		}
		modifier = {
			factor = 4.0
			has_global_flag = 40_CI_MTTH_flag
		}
		modifier = {	#About 15 years for colonies to diverge (this is after the initial establishment which results in about 85 years total for the earlier provinces)
			factor = 0.30
			iberian_old_worlder = yes
		}
		modifier = {
			factor = 0.15	#About 7.5 years if a separate colonial culture
			iberian_new_worlder = yes
		}
		modifier = {	#Once the first provinces flip neighbours will flip faster
			factor = 0.5
			any_neighbor_province = { culture = portuguese_south_african }
		}
		modifier = {	#If independence is achieved divergence is quicker
			factor = 0.4
			AND = {
				culture_group = iberian
				owner = { is_former_colonial_nation = yes }
			}	
		}
	}

	immediate = {
		hidden_effect = {
			every_country = {
				limit = {
					lusophone_cultures_country = yes
					capital_scope = { #This is a fallback trigger in case any new tags spawn after the initial establishment event
						ibero_south_african_culture_region = yes
					}
				}
				change_primary_culture = portuguese_south_african 
			}
		}
	}

	option = {
		name = luso_south_african_settlement.a

		change_culture = portuguese_south_african
		if = {
			limit = { OWNER = { NOT = { accepted_culture = portuguese_south_african } } }
			add_accepted_culture = portuguese_south_african
		}			#Adding culture as an accepted culture to prevent AI flipping, and to not penalise player in a no-choice culture event		
	}
}
