country_event = { #Russka Splinter (Based around the ethnogenesis of Ukrainian)
	id = cultural_influence_east_slavic.1
	title = russka_splinter.title
	desc = russka_splinter.desc
	picture = CITY_VIEW_eventPicture

	major = yes

	trigger = {
		is_year = 1500 #Delay until 1500
		any_owned_province = { #Make sure it actually fires for someone relevant
			OR = {
				culture = ruthenian
				culture = ruthenian_new
			}	
		}
		OR = { #Either condition must be fulfilled
			exists = ZAZ #Zaporozhia was integral to the formation of a modern Ukrainian identity
			LIT = {
				num_of_owned_provinces_with = {
					OR = {
						culture = ruthenian
						culture = ruthenian_new
					}	
					ukrainian_culture_region = yes
					value = 30
				}
			}
			any_country = { #Polish influence on the language
				OR = {
					primary_culture = polish
					primary_culture = polish_new
				}	
				num_of_owned_provinces_with = {
					OR = {
						culture = ruthenian
						culture = ruthenian_new
					}	
					ukrainian_culture_region = yes
					value = 30
				}
			}
			calc_true_if = { #Heavy cossackization
				all_province = {
					ukrainian_culture_region = yes
					OR = {
						culture = cossack
						culture = cossack_new
					}	
				}
				amount = 20
			}
		}
		has_global_flag = CI_events_enabled
		NOT = { has_global_flag = russka_splinter_flag }
	}

	mean_time_to_happen = {
		years = 50

		#MTTH settings
		modifier = {
			factor = 0.0
			has_global_flag = 00_CI_MTTH_flag
		}		
		modifier = {
			factor = 0.5
			has_global_flag = 05_CI_MTTH_flag
		}
		modifier = {
			factor = 1.5
			has_global_flag = 15_CI_MTTH_flag
		}
		modifier = {
			factor = 2.0
			has_global_flag = 20_CI_MTTH_flag
		}
		modifier = {
			factor = 4.0
			has_global_flag = 40_CI_MTTH_flag
		}

		modifier = {
			factor = 0.10 #Near instant
			exists = ZAZ
		}
		modifier = {
			factor = 0.75
			calc_true_if = { #Heavy cossackization
				all_province = {
					ukrainian_culture_region = yes
					OR = {
						culture = cossack
						culture = cossack_new
					}	
				}

			}
		}

			factor = 0.75

					OR = {
						culture = cossack

					OR = {
						culture = cossack
						culture = cossack_new
					}	
				}
				amount = 30
			}
		}

		hidden_effect = {
			#split_russka = yes #Made much more intricate, and is now more based on the ethnogenesis of the Ukrainian state than a divergence (That isn't even that major a fracture, as compared with Frisian or Chagatai)

					culture = ruthenian

country_event = { #Russki Splinter

	title = russki_splinter.title
	desc = russki_splinter.desc
	picture = CITY_VIEW_eventPicture

	major = yes

	trigger = {
		is_year = 1550 #Delay until 1550
		any_owned_province = { #Make sure it actually fires for someone relevant
			OR = {
				culture = russian
				culture = russian_new
			}	
		}
		RUS = { exists = no } #Russia musn't exist
		OR = {
			AND = { #Provinces fell to Foreign rule
				calc_true_if = {
					all_province = {
						OR = {
							culture = russian
							culture = russian_new
						}	
						OR = {
							owner = { NOT = { primary_culture = russian } }
							owner = { NOT = { primary_culture = russian_new } }
						}
					}
					amount = 25
				}
			}
			295 = { #Moscow is under foreign domination (Principally checking for tributary of some nomad)
				OWNER = {
					OR = {
						culture_group = east_slavic
						culture_group = slavic
					}	
					is_subject = yes
					OVERLORD = {
						OR = {
							NOT = { primary_culture = russian }
							NOT = { primary_culture = russian_new }
						}	
					}
				}
			}
			exists = GLH #Golden Horde has formed
			AND = { #Both factional rivals still exist
				exists = MOS
				exists = NOV
				is_year = 1600
			}
		}
		has_global_flag = CI_events_enabled

		}

			factor = 4.0
			has_global_flag = 40_CI_MTTH_flag

			exists = UKR #Ruthenia
