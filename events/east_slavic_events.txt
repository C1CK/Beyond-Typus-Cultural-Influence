country_event = { #Russka Splinter (Based around the ethnogenesis of Ukrainian)
	id = cultural_influence_east_slavic.1
	title = russka_splinter.title
	desc = russka_splinter.desc
	picture = CITY_VIEW_eventPicture

	major = yes

	trigger = {
		is_year = 1500 #Delay until 1500
		any_owned_province = { #Make sure it actually fires for someone relevant
			culture = ruthenian
		}
		OR = { #Either condition must be fulfilled
			exists = ZAZ #Zaporozhia was integral to the formation of a modern Ukrainian identity
			LIT = {
				num_of_owned_provinces_with = {
					culture = ruthenian
					ukrainian_culture_region = yes
					value = 30
				}
			}
			any_country = { #Polish influence on the language
				primary_culture = polish
				num_of_owned_provinces_with = {
					culture = ruthenian
					ukrainian_culture_region = yes
					value = 30
				}
			}
			calc_true_if = { #Heavy cossackization
				all_province = {
					ukrainian_culture_region = yes
					culture = cossack
				}
				amount = 20
			}
		}
		has_global_flag = CI_events_enabled
		NOT = { has_global_flag = russka_splinter_flag }
	}

	mean_time_to_happen = {
		years = 50

		#MTTH settings
		modifier = {
			factor = 0.0
			has_global_flag = 00_CI_MTTH_flag
		}		
		modifier = {
			factor = 0.5
			has_global_flag = 05_CI_MTTH_flag
		}
		modifier = {
			factor = 1.5
			has_global_flag = 15_CI_MTTH_flag
		}
		modifier = {
			factor = 2.0
			has_global_flag = 20_CI_MTTH_flag
		}
		modifier = {
			factor = 4.0
			has_global_flag = 40_CI_MTTH_flag
		}

		modifier = {
			factor = 0.10 #Near instant
			exists = ZAZ
		}
		modifier = {
			factor = 0.75
			calc_true_if = { #Heavy cossackization
				all_province = {
					ukrainian_culture_region = yes
					culture = cossack
				}
				amount = 10
			}
		}
		modifier = {
			factor = 0.75
			calc_true_if = { #Heavy cossackization
				all_province = {
					ukrainian_culture_region = yes
					culture = cossack
				}
				amount = 20
			}
		}
		modifier = {
			factor = 0.75
			calc_true_if = { #Heavy cossackization
				all_province = {
					ukrainian_culture_region = yes
					culture = cossack
				}
				amount = 30
			}
		}
	}

	immediate = {
		hidden_effect = {
			#split_russka = yes #Made much more intricate, and is now more based on the ethnogenesis of the Ukrainian state than a divergence (That isn't even that major a fracture, as compared with Frisian or Chagatai)
			set_global_flag = russka_splinter_flag
			set_global_flag = ukrainian_cossack_identity_flag #Used to tell whether to skip Cossackization in the Ukrainian Heartland

			every_province = {
				limit = {
					culture = ruthenian
					ukrainian_culture_region = yes
					rusyn_culture_region = no #Rusyn takes prio
				}
				change_culture = ukrainian
			}
			every_province = {
				limit = {
					culture = cossack
					ukrainian_culture_region = yes
					ukrainian_cossack_exclusion_zone = no #Areas excluded, primarily the lands traditionally dominated by the Russian Cossacks
				}
				change_culture = ukrainian
				remove_province_modifier = cossack_stanitsa_modifier
			}
			every_province = {
				limit = {
					culture = ruthenian
					ukrainian_culture_region = no #Ukrainian takes prio
					belarussian_culture_region = yes
				}
				change_culture = byelorussian
			}
			every_province = {
				limit = {
					culture = ruthenian
					rusyn_culture_region = yes
				}
				change_culture = rusyn
			}

			every_country = {
				limit = {
					primary_culture = ruthenian
				}
				country_event = {
					id = cultural_influence_east_slavic.2 #Pick new Primary
					days = 0
				}
			}
		}
	}

	option = {
		name = russka_splinter.a
	}
}

country_event = { #Russki Splinter
	id = cultural_influence_east_slavic.3
	title = russki_splinter.title
	desc = russki_splinter.desc
	picture = CITY_VIEW_eventPicture

	major = yes

	trigger = {
		is_year = 1550 #Delay until 1550
		any_owned_province = { #Make sure it actually fires for someone relevant
			culture = russian
		}
		RUS = { exists = no } #Russia musn't exist
		OR = {
			AND = { #Provinces fell to Foreign rule
				calc_true_if = {
					all_province = {
						culture = russian
						owner = { NOT = { primary_culture = russian } }
					}
					amount = 25
				}
			}
			295 = { #Moscow is under foreign domination (Principally checking for tributary of some nomad)
				OWNER = {
					culture_group = east_slavic
					is_subject = yes
					OVERLORD = {
						NOT = { primary_culture = russian }
					}
				}
			}
			exists = GLH #Golden Horde has formed
			AND = { #Both factional rivals still exist
				exists = MOS
				exists = NOV
				is_year = 1600
			}
		}
		has_global_flag = CI_events_enabled
		NOT = { has_global_flag = russki_splinter_flag }
	}

	mean_time_to_happen = {
		years = 50

		#MTTH settings
		modifier = {
			factor = 0.0
			has_global_flag = 00_CI_MTTH_flag
		}		
		modifier = {
			factor = 0.5
			has_global_flag = 05_CI_MTTH_flag
		}
		modifier = {
			factor = 1.5
			has_global_flag = 15_CI_MTTH_flag
		}
		modifier = {
			factor = 2.0
			has_global_flag = 20_CI_MTTH_flag
		}
		modifier = {
			factor = 4.0
			has_global_flag = 40_CI_MTTH_flag
		}

		modifier = {
			factor = 0.5
			exists = UKR #Ruthenia
		}
	}

	immediate = {
		hidden_effect = {
			split_russki = yes

			set_global_flag = russki_splinter_flag

			every_country = {
				limit = {
					primary_culture = russian
				}
				country_event = {
					id = cultural_influence_east_slavic.4 #Pick new Primary
					days = 0
				}
			}
		}
	}

	option = {
		name = russki_spinter.a
	}
}

province_event = { #Northern Russian expansion (Settle) (None of these are valid in the Cossack rule regions)
	id = cultural_influence_east_slavic.7
	title = novgorodian_expansion.title
	desc = novgorodian_expansion.desc
	picture = RELIGIOUS_CONVERSION_eventPicture
	goto = root
	
	trigger = {
		owner = { culture_group = east_slavic }
		northern_russian_culture_region = yes
		cossack_rule_region_european_steppe = no #Needs to be Cossack instead
		cossack_rule_region_siberia = no #Needs to be Cossack instead		
		NOT = { culture_group = east_slavic } #These have special interplay events
		OR = { #Permit AI to settle accepted cultures since they don't understand
			AND = {
				has_owner_accepted_culture = no
				NOT = { culture_group = east_slavic }
			}	
			owner = { ai = yes }
		}
		OR = { #Intent check for beings capable of signaling intent (Requiring this allows lower MTTHs and hopefully more directed efforts)
			AND = { #AIs are dumb (With a lot of work it would be possible to use a system of decisions and flags to signal intent here)
				owner = { ai = yes }
				any_neighbor_province = { culture_group = east_slavic }
			}
			num_of_times_improved_by_owner = 5 #Non-Dharma intent check
			AND = { #Dharma intent check
				has_colonist = yes
				is_city = yes
			}
			AND = { #Passive Settlement Setting
				has_global_flag = ci_passive_settlement_enabled
				any_neighbor_province = { culture_group = east_slavic }
			}				
		}
		OR = { #Players can do it before the decision provided they dev it, or send a colonist
			owner = { has_country_modifier = east_slavic_settlements }
			owner = { has_country_modifier = cossackization }
			owner = { ai = no }
		}
		has_global_flag = CI_events_enabled
	}

	mean_time_to_happen = {
		#Template Settled MTTH
		years = 10

		#MTTH settings
		modifier = {
			factor = 0.0
			has_global_flag = 00_CI_MTTH_flag
		}
		modifier = {
			factor = 0.5
			has_global_flag = 05_CI_MTTH_flag
		}
		modifier = {
			factor = 1.5
			has_global_flag = 15_CI_MTTH_flag
		}
		modifier = {
			factor = 2.0
			has_global_flag = 20_CI_MTTH_flag
		}
		modifier = {
			factor = 4.0
			has_global_flag = 40_CI_MTTH_flag
		}

		#Terrain factors (These replace pre-0.6.1 low dev penalties, which were simulating the same thing) (This was basically done to better simulate the German settlements in Banat)
		#Farmlands & Grasslands are ideal and act as the base 1.0 modifier (Marsh is also treated as Grasslands since it is difficult to tell if the Marsh has been drained)
		modifier = {
			factor = 2.5
			has_terrain = glacier
		}
		modifier = {
			factor = 2
			has_terrain = mountain
		}
		modifier = {
			factor = 1.5
			has_terrain = highlands
		}
		modifier = {
			factor = 1.5
			has_terrain = hills
		}
		modifier = {
			factor = 1.2
			has_terrain = woods
		}
		modifier = {
			factor = 1.2
			has_terrain = drylands
		}
		modifier = {
			factor = 1.4
			has_terrain = forest
		}
		modifier = {
			factor = 2
			has_terrain = desert
		}
		modifier = {
			factor = 1.8
			has_terrain = coastal_desert
		}
		#Climatic factors (Tropical ignored, as it isn't universally difficult to settle, only for the Temperate peoples)
		modifier = {
			has_climate = arid
			factor = 2
		}

		#Dev button factors
		modifier = {
			factor = 0.8
			num_of_times_improved_by_owner = 5
		}
		modifier = {
			factor = 0.8
			num_of_times_improved_by_owner = 10
		}

		#Dev Factors, penalties begin at 30 Dev
		modifier = {
			factor = 1.2
			development = 30
		}
		modifier = {
			factor = 1.2
			development = 40
		}
		modifier = {
			factor = 1.4
			development = 50
		}
		modifier = {
			factor = 1.4
			development = 60
		}
		modifier = {
			factor = 1.4
			development = 70
		}

		#Misc factors
		modifier = { #Passive settlement (I.E without a colonist)
			factor = 3
			has_colonist = no
		}
		modifier = {
			factor = 0.8
			has_idea_to_settle_faster = yes
		}
		modifier = {
			factor = 1.4
			has_owner_religion = no
		}
		modifier = {
			factor = 0.9
			adm = 4
		}
		modifier = {
			factor = 0.9
			adm = 5
		}
		modifier = {
			factor = 0.9
			adm = 6
		}
		#modifier = { #Don't like this, since misc got merged into BT adm focus will be active most of the time now for the advisor cost
		#	factor = 0.75
		#	national_focus = adm
		#}
	}

	option = {
		name = novgorodian_expansion.a

		if = {
			limit = {
				NOT = { has_global_flag = russki_splinter_flag }
			}
			change_culture = russian
		}
		if = {
			limit = {
				has_global_flag = russki_splinter_flag
			}
			change_culture = novgorodian
		}
	}
}
